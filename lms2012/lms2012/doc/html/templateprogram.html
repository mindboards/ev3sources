//        PROGRAM                                                     //

define    MyConstant                    100                           //

vmthread  MAIN                                                        //  void    MAIN(void)
{                                                                     //  {
  DATA8   Run                                                         //
  DATA8   Tmp                                                         //
                                                                      //
  MOVE8_8(1,Run)                                                      //    Run     =  1
                                                                      //    do
Loop:                                                                 //    {
                                                                      //
  MOVE8_8(0,Tmp)                                                      //      Tmp   =  0
  CALL(MySubcall,Tmp)                                                 //      MySubcall(Tmp)
                                                                      //
                                                                      //    }
  JR_TRUE(Run,Loop)                                                   //    while (Run)

}                                                                     //  }
